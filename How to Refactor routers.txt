How to Refactor routers

Example case 1 (simple)

router.get("/preferences-options", async (req, res) => {
  try {
    const composers = await ABCFileModel.distinct("composer");
    const genres = await ABCFileModel.distinct("genre");
    const emotions = await ABCFileModel.distinct("emotion");

    res.status(200).json({ composers, genres, emotions });
  } catch (error) {
    res.status(500).json({ error: "Failed to fetch preferences." });
  }
});

Refactor to route -> controller -> service -> model

1) MOVE LOGIC TO SERVICE FILE

const ABCFileModel = require("../../models/ABCFile"); <-- require model

exports.getMusicPreferencesOptions = async () => { <-- name based on route name
// LOGIC
  const composers = await ABCFileModel.distinct("composer"); 
  const genres = await ABCFileModel.distinct("genre");
  const emotions = await ABCFileModel.distinct("emotion");

  return { composers, genres, emotions };
};

2) HANDLE REQ RES IN CONTROLLER

const userService = require("./user.service"); <-- require service

exports.getMusicPreferencesOptions = async (req, res) => { <-- same name
  try {
    const data = await userService.getMusicPreferencesOptions(); <-- await data from service
    res.status(200).json(data); <-- replace with data
  } catch (error) {
    console.error("Error fetching preferences:", error);
    res.status(500).json({ error: "Failed to fetch preferences." });
  }
};

3) CREATE ROUTE

const router = require("express").Router();
const userController = require("./user.controller"); <-- require controller

router.get("/preferences-options", userController.getMusicPreferencesOptions); <-- route

module.exports = router;

Example case 2 (complex)

router.post("/preferences", async (req, res) => {
  const { composer_preferences, genre_preferences, emotion_preferences } =
    req.body;
  const userId = req.session.userId;
  let client;
  const dbName = "mozartify";
  const collectionName = "users";

  if (!ObjectId.isValid(userId)) {
    return res.status(400).json({ error: "Invalid user ID" });
  }

  try {
    client = new MongoClient(process.env.DB_URI);
    await client.connect();
    const db = client.db(dbName);
    const collectiondb = db.collection(collectionName);

    const updateResult = await collectiondb.updateOne(
      { _id: new ObjectId(userId) },
      {
        $set: {
          composer_preferences,
          genre_preferences,
          emotion_preferences,
          first_timer: false,
        },
      }
    );

    if (updateResult.modifiedCount === 0) {
      return res
        .status(404)
        .json({ error: "User not found or preferences not updated" });
    }

    res.status(200).json({ message: "Preferences updated successfully" });
  } catch (err) {
    console.error(`Error: ${err}`);
    res.status(500).json({ error: "Failed to update preferences" });
  } finally {
    if (client) {
      await client.close();
    }
  }
});

Refactor to route -> controller -> service -> model

1) MOVE LOGIC TO SERVICE FILE, ENSURE REQ AND RES IS HANDLED IN CONTROLLER (note, mongoclient is changed to mongoose for code consistency)

const UserModel = require("../../models/User");
const mongoose = require("mongoose");

exports.updateMusicPreferences = async (userId, preferences) => {
  const { composer_preferences, genre_preferences, emotion_preferences } =
    preferences; <-- changed req.body to preferences which will be passed by the controller

  // Validate userId
  if (!mongoose.Types.ObjectId.isValid(userId)) { <-- changed req.session.userId to userId which will be passed by the controller
    throw new Error("Invalid user ID");
  }

  const updatedUser = await UserModel.findByIdAndUpdate( <-- since not using MongoClient, directly update using Mongoose UserModel
    userId,
    {
      composer_preferences,
      genre_preferences,
      emotion_preferences,
      first_timer: false,
    },
    { new: true }
  );

  if (!updatedUser) { <-- change accordingly
    throw new Error("User not found or preferences not updated");
  }

  return updatedUser; <-- replace res with return update results, service returns data, controller decides response
};

2) HANDLE ALL REQ RES IN CONTROLLER

const userService = require("./user.service");

exports.updateMusicPreferences = async (req, res) => {
  try {
    const userId = req.session.userId; <-- the req.session.userID is placed here

    await userService.updateMusicPreferences(userId, req.body); <-- the req.body for preferences is placed here

    res.status(200).json({
      message: "Preferences updated successfully",
    });
  } catch (err) {
    console.error("Error updating preferences:", err.message);

    if (err.message === "Invalid user ID") { <-- err.message comes from service Error() method
      return res.status(400).json({ error: err.message }); <-- return res status 400 according to the message
    }

    if (err.message.includes("not found")) {
      return res.status(404).json({ error: err.message });
    }

    res.status(500).json({ error: "Failed to update preferences" });
  }
};

3) CREATE ROUTE

const router = require("express").Router();
const userController = require("./user.controller");
const { isAuthenticated } = require("../auth/middleware/auth.middleware");

router.post("/preferences", isAuthenticated, userController.updateMusicPreferences); <-- isAuthenticated middleware is needed due to the use of req.session

module.exports = router;
